# Runtime dependencies for teac-dab-controls.

# Volumio websocket client. Must stay on the socketio 4.x / engineio 3.x line
# to remain compatible with Volumio's socket.io server protocol.

python-socketio[client]>=4,<5

pigpio
spidev

# REST API server
fastapi
uvicorn

# Retry helper used by the Volumio client
retrying
